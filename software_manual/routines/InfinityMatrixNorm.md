    def matrixNormInfinity(A):
        n = len(A)
        maxSum = 0
        for i in range(n):
            newSum = 0
            for j in range(n):
                newSum += abs(A[i][j])

            if newSum > maxSum:
                maxSum = newSum

        return maxSum
