    # takes a matrix and returns the sum of the column with greatest magnitude
    def matrixNormL1(A):
        n = len(A)
        maxSum = 0
        for i in range(n):
            newSum = 0
            for j in range(n):
                newSum += abs(A[j][i])

            if newSum > maxSum:
                maxSum = newSum

        return maxSum
